cmake_minimum_required(VERSION 3.14)
project(Practice03_Tests)

# Enable testing framework
enable_testing()

# Fetch Google Test
include(FetchContent)
FetchContent_Declare(
    googletest
    URL https://github.com/google/googletest/archive/refs/tags/v1.14.0.zip
)
FetchContent_MakeAvailable(googletest)

# Add the test executable
add_executable(practice03_test practice03_test/practice03_test.cpp)

# Link Google Test
target_link_libraries(practice03_test PRIVATE gtest gtest_main)

# Register tests
add_test(NAME Practice03Test COMMAND practice03_test)
